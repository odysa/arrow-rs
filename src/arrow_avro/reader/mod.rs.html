18//! Avro reader
19//!
20//! This module provides facilities to read Apache Avro-encoded files or streams
21//! into Arrow's `RecordBatch` format. In particular, it introduces:
22//!
23//! * `ReaderBuilder`: Configures Avro reading, e.g., batch size
24//! * `Reader`: Yields `RecordBatch` values, implementing `Iterator`
25//! * `Decoder`: A low-level push-based decoder for Avro records
26//!
27//! # Basic Usage
28//!
29//! `Reader` can be used directly with synchronous data sources, such as [`std::fs::File`].
30//!
31//! ## Reading a Single Batch
32//!
33//! ```
34//! # use std::fs::File;
35//! # use std::io::BufReader;
36//! # use arrow_avro::reader::ReaderBuilder;
37//!
38//! let file = File::open("../testing/data/avro/alltypes_plain.avro").unwrap();
39//! let mut avro = ReaderBuilder::new().build(BufReader::new(file)).unwrap();
40//! let batch = avro.next().unwrap();
41//! ```
42//!
43//! # Async Usage
44//!
45//! The lower-level `Decoder` can be integrated with various forms of async data streams,
46//! and is designed to be agnostic to different async IO primitives within
47//! the Rust ecosystem. It works by incrementally decoding Avro data from byte slices.
48//!
49//! For example, see below for how it could be used with an arbitrary `Stream` of `Bytes`:
50//!
51//! ```
52//! # use std::task::{Poll, ready};
53//! # use bytes::{Buf, Bytes};
54//! # use arrow_schema::ArrowError;
55//! # use futures::stream::{Stream, StreamExt};
56//! # use arrow_array::RecordBatch;
57//! # use arrow_avro::reader::Decoder;
58//!
59//! fn decode_stream<S: Stream<Item = Bytes> + Unpin>(
60//!     mut decoder: Decoder,
61//!     mut input: S,
62//! ) -> impl Stream<Item = Result<RecordBatch, ArrowError>> {
63//!     let mut buffered = Bytes::new();
64//!     futures::stream::poll_fn(move |cx| {
65//!         loop {
66//!             if buffered.is_empty() {
67//!                 buffered = match ready!(input.poll_next_unpin(cx)) {
68//!                     Some(b) => b,
69//!                     None => break,
70//!                 };
71//!             }
72//!             let decoded = match decoder.decode(buffered.as_ref()) {
73//!                 Ok(decoded) => decoded,
74//!                 Err(e) => return Poll::Ready(Some(Err(e))),
75//!             };
76//!             let read = buffered.len();
77//!             buffered.advance(decoded);
78//!             if decoded != read {
79//!                 break
80//!             }
81//!         }
82//!         // Convert any fully-decoded rows to a RecordBatch, if available
83//!         Poll::Ready(decoder.flush().transpose())
84//!     })
85//! }
86//! ```
87//!

104/// Read the Avro file header (magic, metadata, sync marker) from `reader`.
105fn read_header<R: BufRead>(mut reader: R) -> Result<Header, ArrowError> {
106    let mut decoder = HeaderDecoder::default();
107    loop {
108        let buf = reader.fill_buf()?;
109        if buf.is_empty() {
110            break;
111        }
112        let read = buf.len();
113        let decoded = decoder.decode(buf)?;
114        reader.consume(decoded);
115        if decoded != read {
116            break;
117        }
118    }
119    decoder.flush().ok_or_else(|| {
120        ArrowError::ParseError("Unexpected EOF while reading Avro header".to_string())
121    })
122}

124/// A low-level interface for decoding Avro-encoded bytes into Arrow `RecordBatch`.
125#[derive(Debug)]
126pub struct Decoder {
127    record_decoder: RecordDecoder,
128    batch_size: usize,
129    decoded_rows: usize,
130}
131
132impl Decoder {
133    fn new(record_decoder: RecordDecoder, batch_size: usize) -> Self {
134        Self {
135            record_decoder,
136            batch_size,
137            decoded_rows: 0,
138        }
139    }
140
141    /// Return the Arrow schema for the rows decoded by this decoder
142    pub fn schema(&self) -> SchemaRef {
143        self.record_decoder.schema().clone()
144    }
145
146    /// Return the configured maximum number of rows per batch
147    pub fn batch_size(&self) -> usize {
148        self.batch_size
149    }
150
151    /// Feed `data` into the decoder row by row until we either:
152    /// - consume all bytes in `data`, or
153    /// - reach `batch_size` decoded rows.
154    ///
155    /// Returns the number of bytes consumed.
156    pub fn decode(&mut self, data: &[u8]) -> Result<usize, ArrowError> {
157        let mut total_consumed = 0usize;
158        while total_consumed < data.len() && self.decoded_rows < self.batch_size {
159            let consumed = self.record_decoder.decode(&data[total_consumed..], 1)?;
160            if consumed == 0 {
161                break;
162            }
163            total_consumed += consumed;
164            self.decoded_rows += 1;
165        }
166        Ok(total_consumed)
167    }
168
169    /// Produce a `RecordBatch` if at least one row is fully decoded, returning
170    /// `Ok(None)` if no new rows are available.
171    pub fn flush(&mut self) -> Result<Option<RecordBatch>, ArrowError> {
172        if self.decoded_rows == 0 {
173            Ok(None)
174        } else {
175            let batch = self.record_decoder.flush()?;
176            self.decoded_rows = 0;
177            Ok(Some(batch))
178        }
179    }
180
181    /// Returns the number of rows that can be added to this decoder before it is full.
182    pub fn capacity(&self) -> usize {
183        self.batch_size.saturating_sub(self.decoded_rows)
184    }
185
186    /// Returns true if the decoder has reached its capacity for the current batch.
187    pub fn batch_is_full(&self) -> bool {
188        self.capacity() == 0
189    }
190}

192/// A builder to create an [`Avro Reader`](Reader) that reads Avro data
193/// into Arrow `RecordBatch`.
194#[derive(Debug)]
195pub struct ReaderBuilder {
196    batch_size: usize,
197    strict_mode: bool,
198    utf8_view: bool,
199    schema: Option<AvroSchema<'static>>,
200}
201
202impl Default for ReaderBuilder {
203    fn default() -> Self {
204        Self {
205            batch_size: 1024,
206            strict_mode: false,
207            utf8_view: false,
208            schema: None,
209        }
210    }
211}
212
213impl ReaderBuilder {
214    /// Creates a new [`ReaderBuilder`] with default settings:
215    /// - `batch_size` = 1024
216    /// - `strict_mode` = false
217    /// - `utf8_view` = false
218    /// - `schema` = None
219    pub fn new() -> Self {
220        Self::default()
221    }
222
223    fn make_record_decoder(&self, schema: &AvroSchema<'_>) -> Result<RecordDecoder, ArrowError> {
224        let root_field = AvroFieldBuilder::new(schema)
225            .with_utf8view(self.utf8_view)
226            .with_strict_mode(self.strict_mode)
227            .build()?;
228        RecordDecoder::try_new_with_options(root_field.data_type(), self.utf8_view)
229    }
230
231    fn build_impl<R: BufRead>(self, reader: &mut R) -> Result<(Header, Decoder), ArrowError> {
232        let header = read_header(reader)?;
233        let record_decoder = if let Some(schema) = &self.schema {
234            self.make_record_decoder(schema)?
235        } else {
236            let avro_schema: Option<AvroSchema<'_>> = header
237                .schema()
238                .map_err(|e| ArrowError::ExternalError(Box::new(e)))?;
239            let avro_schema = avro_schema.ok_or_else(|| {
240                ArrowError::ParseError("No Avro schema present in file header".to_string())
241            })?;
242            self.make_record_decoder(&avro_schema)?
243        };
244        let decoder = Decoder::new(record_decoder, self.batch_size);
245        Ok((header, decoder))
246    }
247
248    /// Sets the row-based batch size
249    pub fn with_batch_size(mut self, batch_size: usize) -> Self {
250        self.batch_size = batch_size;
251        self
252    }
253
254    /// Set whether to use StringViewArray for string data
255    ///
256    /// When enabled, string data from Avro files will be loaded into
257    /// Arrow's StringViewArray instead of the standard StringArray.
258    pub fn with_utf8_view(mut self, utf8_view: bool) -> Self {
259        self.utf8_view = utf8_view;
260        self
261    }
262
263    /// Get whether StringViewArray is enabled for string data
264    pub fn use_utf8view(&self) -> bool {
265        self.utf8_view
266    }
267
268    /// Controls whether certain Avro unions of the form `[T, "null"]` should produce an error.
269    pub fn with_strict_mode(mut self, strict_mode: bool) -> Self {
270        self.strict_mode = strict_mode;
271        self
272    }
273
274    /// Sets the Avro schema.
275    ///
276    /// If a schema is not provided, the schema will be read from the Avro file header.
277    pub fn with_schema(mut self, schema: AvroSchema<'static>) -> Self {
278        self.schema = Some(schema);
279        self
280    }
281
282    /// Create a [`Reader`] from this builder and a `BufRead`
283    pub fn build<R: BufRead>(self, mut reader: R) -> Result<Reader<R>, ArrowError> {
284        let (header, decoder) = self.build_impl(&mut reader)?;
285        Ok(Reader {
286            reader,
287            header,
288            decoder,
289            block_decoder: BlockDecoder::default(),
290            block_data: Vec::new(),
291            block_cursor: 0,
292            finished: false,
293        })
294    }
295
296    /// Create a [`Decoder`] from this builder and a `BufRead` by
297    /// reading and parsing the Avro file's header. This will
298    /// not create a full [`Reader`].
299    pub fn build_decoder<R: BufRead>(self, mut reader: R) -> Result<Decoder, ArrowError> {
300        match self.schema {
301            Some(ref schema) => {
302                let record_decoder = self.make_record_decoder(schema)?;
303                Ok(Decoder::new(record_decoder, self.batch_size))
304            }
305            None => {
306                let (_, decoder) = self.build_impl(&mut reader)?;
307                Ok(decoder)
308            }
309        }
310    }
311}

313/// A high-level Avro `Reader` that reads container-file blocks
314/// and feeds them into a row-level [`Decoder`].
315#[derive(Debug)]
316pub struct Reader<R: BufRead> {
317    reader: R,
318    header: Header,
319    decoder: Decoder,
320    block_decoder: BlockDecoder,
321    block_data: Vec<u8>,
322    block_cursor: usize,
323    finished: bool,
324}
325
326impl<R: BufRead> Reader<R> {
327    /// Return the Arrow schema discovered from the Avro file header
328    pub fn schema(&self) -> SchemaRef {
329        self.decoder.schema()
330    }
331
332    /// Return the Avro container-file header
333    pub fn avro_header(&self) -> &Header {
334        &self.header
335    }
336
337    /// Reads the next [`RecordBatch`] from the Avro file or `Ok(None)` on EOF
338    fn read(&mut self) -> Result<Option<RecordBatch>, ArrowError> {
339        'outer: while !self.finished && !self.decoder.batch_is_full() {
340            while self.block_cursor == self.block_data.len() {
341                let buf = self.reader.fill_buf()?;
342                if buf.is_empty() {
343                    self.finished = true;
344                    break 'outer;
345                }
346                // Try to decode another block from the buffered reader.
347                let consumed = self.block_decoder.decode(buf)?;
348                self.reader.consume(consumed);
349                if let Some(block) = self.block_decoder.flush() {
350                    // Successfully decoded a block.
351                    let block_data = if let Some(ref codec) = self.header.compression()? {
352                        codec.decompress(&block.data)?
353                    } else {
354                        block.data
355                    };
356                    self.block_data = block_data;
357                    self.block_cursor = 0;
358                } else if consumed == 0 {
359                    // The block decoder made no progress on a non-empty buffer.
360                    return Err(ArrowError::ParseError(
361                        "Could not decode next Avro block from partial data".to_string(),
362                    ));
363                }
364            }
365            // Try to decode more rows from the current block.
366            let consumed = self.decoder.decode(&self.block_data[self.block_cursor..])?;
367            if consumed == 0 && self.block_cursor < self.block_data.len() {
368                self.block_cursor = self.block_data.len();
369            } else {
370                self.block_cursor += consumed;
371            }
372        }
373        self.decoder.flush()
374    }
375}
376
377impl<R: BufRead> Iterator for Reader<R> {
378    type Item = Result<RecordBatch, ArrowError>;
379
380    fn next(&mut self) -> Option<Self::Item> {
381        self.read().transpose()
382    }
383}

385impl<R: BufRead> RecordBatchReader for Reader<R> {
386    fn schema(&self) -> SchemaRef {
387        self.schema()
388    }
389}

416    fn read_file(path: &str, batch_size: usize, utf8_view: bool) -> RecordBatch {
417        let file = File::open(path).unwrap();
418        let reader = ReaderBuilder::new()
419            .with_batch_size(batch_size)
420            .with_utf8_view(utf8_view)
421            .build(BufReader::new(file))
422            .unwrap();
423        let schema = reader.schema();
424        let batches = reader.collect::<Result<Vec<_>, _>>().unwrap();
425        arrow::compute::concat_batches(&schema, &batches).unwrap()
426    }

441    fn decode_stream<S: Stream<Item = Bytes> + Unpin>(
442        mut decoder: Decoder,
443        mut input: S,
444    ) -> impl Stream<Item = Result<RecordBatch, ArrowError>> {
445        async_stream::try_stream! {
446            if let Some(data) = input.next().await {
447                let consumed = decoder.decode(&data)?;
448                if consumed < data.len() {
449                    Err(ArrowError::ParseError(
450                        "did not consume all bytes".to_string(),
451                    ))?;
452                }
453            }
454            if let Some(batch) = decoder.flush()? {
455                yield batch
456            }
457        }
458    }

604        struct TestCase<'a> {
605            name: &'a str,
606            schema: &'a str,
607            expected_error: Option<&'a str>,
608        }
609        let tests = vec![
610            TestCase {
611                name: "success",

616                name: "valid schema invalid data",
617                schema: r#"{"type":"record","name":"test","fields":[{"name":"f2","type":"long"}]}"#,

619            },
620        ];
621        for test in tests {
622            let schema_s2: crate::schema::Schema = serde_json::from_str(test.schema).unwrap();
623            let record_val = "some_string";
624            let mut body = vec![];
625            body.push((record_val.len() as u8) << 1);
626            body.extend_from_slice(record_val.as_bytes());
627            let mut reader_placeholder = Cursor::new(&[] as &[u8]);
628            let builder = ReaderBuilder::new()
629                .with_batch_size(1)
630                .with_schema(schema_s2);
631            let decoder_result = builder.build_decoder(&mut reader_placeholder);
632            let decoder = match decoder_result {
633                Ok(decoder) => decoder,
634                Err(e) => {
635                    if let Some(expected) = test.expected_error {
636                        assert!(
637                            e.to_string().contains(expected),
638                            "Test '{}' failed: unexpected error message at build.\nExpected to contain: '{expected}'\nActual: '{e}'",
639                            test.name,
640                        );
641                        continue;
642                    } else {
643                        panic!("Test '{}' failed at decoder build: {e}", test.name);
644                    }
645                }
646            };
647            let stream = Box::pin(stream::once(async { Bytes::from(body) }));
648            let decoded_stream = decode_stream(decoder, stream);
649            let batches_result: Result<Vec<RecordBatch>, ArrowError> =
650                block_on(decoded_stream.try_collect());
651            match (batches_result, test.expected_error) {
652                (Ok(batches), None) => {

654                        arrow::compute::concat_batches(&batches[0].schema(), &batches).unwrap();
655                    let expected_field = Field::new("f2", DataType::Utf8, false);
656                    let expected_schema = Arc::new(Schema::new(vec![expected_field]));
657                    let expected_array = Arc::new(StringArray::from(vec![record_val]));
658                    let expected_batch =
659                        RecordBatch::try_new(expected_schema, vec![expected_array]).unwrap();
660                    assert_eq!(batch, expected_batch, "Test '{}' failed", test.name);
661                    assert_eq!(
662                        batch.schema().field(0).name(),
663                        "f2",
664                        "Test '{}' failed",
665                        test.name
666                    );
667                }
668                (Err(e), Some(expected)) => {
669                    assert!(
